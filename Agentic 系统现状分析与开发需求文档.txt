```python
# 新增 task_scheduler.py

from collections import defaultdict, deque

class TaskGraph:
    """任务依赖图"""
    def __init__(self):
        self.tasks = {}  # task_id -> task_info
        self.dependencies = defaultdict(list)  # task_id -> [dependent_ids]
        self.reverse_deps = defaultdict(list)  # task_id -> [prerequisite_ids]

    def add_task(self, task_id: str, role: str, task: str, depends_on: List[str]):
        self.tasks[task_id] = {
            "id": task_id,
            "role": role,
            "task": task,
            "status": "pending",
            "result": None
        }
        self.reverse_deps[task_id] = depends_on
        for dep in depends_on:
            self.dependencies[dep].append(task_id)

    def get_ready_tasks(self) -> List[str]:
        """获取所有可执行的任务（依赖已满足）"""
        ready = []
        for task_id, task in self.tasks.items():
            if task["status"] != "pending":
                continue
            prereqs = self.reverse_deps[task_id]
            if all(self.tasks[p]["status"] == "completed" for p in prereqs):
                ready.append(task_id)
        return ready

    def is_complete(self) -> bool:
        return all(t["status"] in ["completed", "failed"] for t in self.tasks.values())

# SupervisorAgent 改造
async def execute_with_dag(self, query: str, llm_client):
    # 1. 获取任务分解
    sub_tasks = await self.decompose_task_with_llm(query, intent, llm_client)

    # 2. 构建任务图
    graph = TaskGraph()
    for i, t in enumerate(sub_tasks):
        graph.add_task(
            task_id=f"task_{i}",
            role=t["role"],
            task=t["task"],
            depends_on=t.get("depends_on", [])
        )

    # 3. 执行调度循环
    while not graph.is_complete():
        ready_tasks = graph.get_ready_tasks()
        if not ready_tasks:
            # 检测死锁或失败
            break

        # 并行执行就绪任务
        results = await asyncio.gather(
            *[self._execute_task(graph.tasks[tid], llm_client)
              for tid in ready_tasks],
            return_exceptions=True
        )

        # 更新任务状态
        for tid, result in zip(ready_tasks, results):
            if isinstance(result, Exception):
                graph.tasks[tid]["status"] = "failed"
                graph.tasks[tid]["error"] = str(result)
            else:
                graph.tasks[tid]["status"] = "completed"
                graph.tasks[tid]["result"] = result
```

```python
# 新增 mcp/client.py

from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client
from mcp.client.sse import sse_client

class MCPClient:
    def __init__(self, server_config: dict):
        self.config = server_config
        self.session = None
        self.tools = []

    async def connect(self):
        """连接到 MCP 服务器"""
        if self.config["connection_type"] == "stdio":
            # stdio 连接
            params = StdioServerParameters(
                command=self.config["command"].split()[0],
                args=self.config["command"].split()[1:],
                env=self.config.get("env_vars", {})
            )
            transport = await stdio_client(params).__aenter__()
        elif self.config["connection_type"] == "sse":
            # SSE 连接
            transport = await sse_client(self.config["url"]).__aenter__()

        reader, writer = transport
        self.session = ClientSession(reader, writer)
        await self.session.initialize()

        # 获取工具列表
        response = await self.session.list_tools()
        self.tools = response.tools

    async def call_tool(self, tool_name: str, arguments: dict):
        """调用工具"""
        result = await self.session.call_tool(tool_name, arguments)
        return result

    async def disconnect(self):
        if self.session:
            await self.session.__aexit__(None, None, None)

# routes/mcp.py 改造
```

```python
# 新增 tools/executor.py

from typing import Any, Dict, Optional
import jsonschema
import asyncio

class ToolExecutor:
    def __init__(self, db_session):
        self.db = db_session
        self.mcp_clients = {}  # server_id -> MCPClient

    async def get_or_create_client(self, server_id: str) -> MCPClient:
        """获取或创建 MCP 客户端"""
        if server_id not in self.mcp_clients:
            # 从数据库加载服务器配置
            result = await self.db.execute(
                select(MCPServer).where(MCPServer.id == server_id)
            )
            server = result.scalar_one()
            self.mcp_clients[server_id] = MCPClient(server.to_dict())
            await self.mcp_clients[server_id].connect()
        return self.mcp_clients[server_id]

    async def execute_tool(
        self,
        tool_id: str,
        arguments: Dict[str, Any],
        timeout: float = 30.0
    ) -> Dict:
        """执行工具调用"""
        # 1. 获取工具信息
        result = await self.db.execute(select(Tool).where(Tool.id == tool_id))
        tool = result.scalar_one()

        if not tool:
            raise ValueError(f"Tool {tool_id} not found")

        if not tool.is_enabled:
            raise ValueError(f"Tool {tool.name} is disabled")

        # 2. 参数验证（使用 JSON Schema）
        if tool.config_schema:
            try:
                jsonschema.validate(arguments, tool.config_schema)
            except jsonschema.ValidationError as e:
                raise ValueError(f"Invalid parameters: {e.message}")

        # 3. 执行调用（带超时）
        start_time = datetime.utcnow()
        try:
            if tool.is_mcp:
                client = await self.get_or_create_client(tool.mcp_server_id)
                result = await asyncio.wait_for(
                    client.call_tool(tool.name, arguments),
                    timeout=timeout
                )
            else:
                # 内置工具调用
                result = await self._call_builtin_tool(tool, arguments)

            duration = (datetime.utcnow() - start_time).total_seconds()

            # 4. 记录日志
            log = ToolCallLog(
                tool_id=tool_id,
                status="success",
                result=result,
                duration=duration,
                params=arguments
            )
            self.db.add(log)

            # 5. 更新统计
            tool.usage_count += 1
            tool.success_rate = self._calculate_success_rate(tool_id)

            return {
                "success": True,
                "result": result,
                "duration": duration
            }

        except asyncio.TimeoutError:
            # 记录超时日志
            log = ToolCallLog(
                tool_id=tool_id,
                status="timeout",
                error_message=f"Timeout after {timeout}s",
                duration=timeout,
                params=arguments
            )
            self.db.add(log)
            raise

        except Exception as e:
            # 记录错误日志
            log = ToolCallLog(
                tool_id=tool_id,
                status="error",
                error_message=str(e),
                duration=(datetime.utcnow() - start_time).total_seconds(),
                params=arguments
            )
```

```python
            if isinstance(node, ast.Num):
                return node.n
            elif isinstance(node, ast.BinOp):
                left = eval_expr(node.left)
                right = eval_expr(node.right)
                return operators[type(node.op)](left, right)
            else:
                raise ValueError("Unsupported operation")

        tree = ast.parse(expression, mode='eval')
        result = eval_expr(tree.body)

        return {"result": result, "expression": expression}
```

#### 4.2 实现错误恢复和重试机制
**要干的内容：**
- 检测工具调用失败
- 自动重试（最多 3 次）
- 降级方案（使用备用工具或跳过）

**改造细节：**
```python
# tools/executor.py 增强

class ToolExecutor:
    async def execute_tool_with_retry(
        self,
        tool_id: str,
        arguments: Dict[str, Any],
        max_retries: int = 3,
        timeout: float = 30.0
    ) -> Dict:
        """带重试的工具调用"""
        last_error = None

        for attempt in range(max_retries):
            try:
                result = await self.execute_tool(tool_id, arguments, timeout)

                # 检查结果是否有效
                if self._is_valid_result(result):
                    return result
                else:
                    raise ValueError("Invalid result format")

            except asyncio.TimeoutError as e:
                last_error = e
                print(f"[ToolExecutor] 超时重试 {attempt + 1}/{max_retries}")
                await asyncio.sleep(2 ** attempt)  # 指数退避

            except Exception as e:
                last_error = e
                print(f"[ToolExecutor] 错误重试 {attempt + 1}/{max_retries}: {e}")

                # 如果是参数错误，不重试
                if "Invalid parameters" in str(e):
                    raise

                await asyncio.sleep(2 ** attempt)

        # 所有重试失败，尝试降级方案
        return await self._fallback_execute(tool_id, arguments, last_error)

    async def _fallback_execute(self, tool_id: str, arguments: Dict, original_error: Exception) -> Dict:
        """降级执行"""
        # 1. 查找替代工具
        alternative_tools = await self._find_alternative_tools(tool_id)

        for alt_tool in alternative_tools:
            try:
                return await self.execute_tool(alt_tool.id, arguments)
            except Exception:
                continue

        # 2. 如果无替代工具，返回错误但标记为可恢复
        return {
            "success": False,
            "error": str(original_error),
            "recoverable": True,
            "suggestion": "尝试简化查询或使用其他工具"
        }
```

```javascript
// frontend/src/components/ToolSelector.jsx

import React, { useState, useMemo } from 'react';
import { Dialog } from '@headlessui/react';
import { useStore } from '../store';

const ToolSelector = ({ isOpen, onClose }) => {
  const { tools, selectedTools, toggleTool } = useStore();
  const [searchQuery, setSearchQuery] = useState('');
  const [expandedCategories, setExpandedCategories] = useState(['常用']);

  // 按分类分组工具
  const groupedTools = useMemo(() => {
    const groups = {};
    tools.forEach(tool => {
      const category = tool.category || '其他';
      if (!groups[category]) groups[category] = [];
      groups[category].push(tool);
    });
    return groups;
  }, [tools]);

  // 搜索过滤
  const filteredGroups = useMemo(() => {
    if (!searchQuery) return groupedTools;

    const result = {};
    Object.entries(groupedTools).forEach(([category, categoryTools]) => {
      const filtered = categoryTools.filter(tool =>
        tool.name.toLowerCase().includes(searchQuery.toLowerCase()) ||
        tool.description.toLowerCase().includes(searchQuery.toLowerCase())
      );
      if (filtered.length > 0) result[category] = filtered;
    });
    return result;
  }, [groupedTools, searchQuery]);

  const toggleCategory = (category) => {
    setExpandedCategories(prev =>
      prev.includes(category)
        ? prev.filter(c => c !== category)
        : [...prev, category]
    );
  };

  const enableAll = () => {
    Object.values(filteredGroups).flat().forEach(tool => {
      if (!selectedTools.includes(tool.id)) toggleTool(tool.id);
    });
  };

  const disableAll = () => {
    Object.values(filteredGroups).flat().forEach(tool => {
      if (selectedTools.includes(tool.id)) toggleTool(tool.id);
    });
  };

  return (
    <Dialog open={isOpen} onClose={onClose} className="fixed inset-0 z-50">
      <div className="flex items-center justify-center min-h-screen px-4">
        <Dialog.Overlay className="fixed inset-0 bg-black opacity-30" />

        <div className="bg-white rounded-lg max-w-2xl w-full max-h-[600px] flex flex-col relative z-10">
          {/* 标题栏 */}
          <div className="p-4 border-b flex items-center justify-between">
            <Dialog.Title className="font-bold">选择工具</Dialog.Title>
            <button onClick={onClose}>✕</button>
          </div>

          {/* 搜索框 */}
          <div className="p-4 border-b">
            <input
              type="text"
              placeholder="搜索工具..."
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
              className="w-full border rounded px-3 py-2"
            />
          </div>

          {/* 工具列表 */}
          <div className="flex-1 overflow-y-auto p-4">
            {Object.entries(filteredGroups).map(([category, categoryTools]) => (
              <div key={category} className="mb-4">
                <button
                  onClick={() => toggleCategory(category)}
                  className="flex items-center justify-between w-full text-left font-medium mb-2"
                >
                  <span>{getCategoryIcon(category)} {category}</span>
                  <span>{expandedCategories.includes(category) ? '▼' : '▶'}</span>
                </button>

                {expandedCategories.includes(category) && (
                  <div className="grid grid-cols-3 gap-3">
                    {categoryTools.map(tool => (
                      <div
                        key={tool.id}
                        className={`border rounded-lg p-3 cursor-pointer transition-colors ${
                          selectedTools.includes(tool.id)
                            ? 'bg-blue-50 border-blue-300'
                            : 'hover:bg-gray-50'
                        }`}
                        onClick={() => toggleTool(tool.id)}
                      >
                        <div className="text-2xl mb-1">{tool.icon || '🛠️'}</div>
                        <div className="text-sm font-medium truncate">{tool.name}</div>
                        <div className="text-xs text-gray-500 line-clamp-2 h-8">
                          {tool.description}
                        </div>
                        <div className="mt-2 flex items-center justify-between">
                          <div className={`w-3 h-3 rounded-full ${
                            selectedTools.includes(tool.id) ? 'bg-blue-500' : 'bg-gray-300'
                          }`} />
                          <div className="text-xs text-gray-400">
                            {tool.is_mcp ? 'MCP' : '内置'}
                          </div>
                        </div>
                      </div>
                    ))}
                  </div>
                )}
              </div>
            ))}
          </div>

          {/* 底部操作栏 */}
          <div className="p-4 border-t flex items-center justify-between">
            <div className="flex gap-2">
              <button onClick={enableAll} className="text-sm text-blue-500">
                批量启用全部
              </button>
              <button onClick={disableAll} className="text-sm text-red-500">
                批量禁用全部
              </button>
            </div>
            <button
              onClick={onClose}
              className="px-4 py-2 bg-blue-500 text-white rounded hover:bg-blue-600"
            >
              确认
            </button>
          </div>
        </div>
      </div>
    </Dialog>
  );
};

export default ToolSelector;
```

**验收标准：**
- 工具按分类展示
- 搜索功能正常
- 点击工具切换选中状态
- 批量操作生效

```python
# routes/sessions.py 新增

from fastapi.responses import Response
import markdown
from reportlab.lib.pagesizes import letter
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer
from io import BytesIO

@router.get("/{session_id}/export")
async def export_session(
    session_id: str,
    format: str = "markdown",
    db: AsyncSession = Depends(get_db)
):
    """导出会话"""
    # 获取会话和消息
    result = await db.execute(
        select(Session).where(Session.id == session_id)
    )
    session = result.scalar_one()

    messages = session.messages

    if format == "markdown":
        content = generate_markdown(session, messages)
        return Response(
            content=content,
            media_type="text/markdown",
            headers={"Content-Disposition": f"attachment; filename=session_{session_id}.md"}
        )

    elif format == "pdf":
        buffer = generate_pdf(session, messages)
        return Response(
            content=buffer.getvalue(),
            media_type="application/pdf",
            headers={"Content-Disposition": f"attachment; filename=session_{session_id}.pdf"}
        )

    elif format == "json":
        return {
            "session_id": session.id,
            "title": session.title,
            "created_at": session.created_at.isoformat(),
            "messages": [
                {
                    "role": msg.role,
                    "content": msg.content,
                    "timestamp": msg.timestamp.isoformat(),
                    "thinking_process": msg.thinking_process,
                    "sub_agent_results": msg.sub_agent_results
                }
                for msg in messages
            ]
        }

def generate_markdown(session, messages):
    """生成 Markdown 格式"""
    md = f"# {session.title}\n\n"
    md += f"**创建时间**: {session.created_at.strftime('%Y-%m-%d %H:%M:%S')}\n\n"
    md += "---\n\n"

    for msg in messages:
        if msg.role == "user":
            md += f"## 👤 用户\n\n{msg.content}\n\n"
        else:
            md += f"## 🤖 AI\n\n"

            # 思考过程
            if msg.thinking_process:
                md += "### 🧠 思考过程\n\n"
                for step in msg.thinking_process:
                    md += f"- **{step['agent']}**: {step['action']} ({step['duration']:.1f}s)\n"
                md += "\n"

            # 子智能体结果
            if msg.sub_agent_results:
                md += "### 🤖 子智能体协作\n\n"
                for agent in msg.sub_agent_results:
                    md += f"#### {agent['role']}\n\n"
                    md += f"- 任务：{agent['task']}\n"
                    md += f"- 耗时：{agent['duration']:.1f}s\n"
                    md += f"- 输出：{agent['output']}\n\n"

            # 最终答案
            md += f"### 答案\n\n{msg.content}\n\n"

        md += "---\n\n"

    return md
```

## 第四部分：技术债务和风险

### 已知技术债务
1. **数据库迁移**：当前使用 SQLite，生产环境需切换到 PostgreSQL
2. **认证授权**：缺少用户认证和权限控制
3. **并发控制**：多用户同时访问时的资源竞争问题
4. **监控告警**：缺少系统监控和异常告警

### 潜在风险
1. **MCP SDK 兼容性**：官方 SDK 可能不稳定
2. **LLM 成本**：频繁调用 LLM 可能导致成本过高
3. **性能瓶颈**：复杂任务可能导致响应时间过长
4. **安全性**：工具调用可能存在注入攻击风险

### 缓解措施
- 逐步替换模拟实现，每个模块充分测试后再上线
- 实现 LLM 响应缓存，减少重复调用
- 设置任务超时限制和资源配额
- 对所有工具输入进行严格验证
